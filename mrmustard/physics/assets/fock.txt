    plot_fig = False
    if len(rep.array.shape) == 2:
        # `n` on the `x` axis, batch on the `y` axis
        plot_fig = True
        is_batched = True
        array = rep.array
    elif len(rep.array.shape) == 3 and rep.array.shape[0] == 1:
        # `n0` on the `x` axis, `n1` on the `y`
        plot_fig = True
        is_batched = False
        array = rep.array[0]

    if plot_fig:
        text = []
        for y, row in enumerate(array):
            text.append([])
            for x, val in enumerate(row):
                info = f"batch: {y}<br />" if is_batched else f"axis 0: {y}<br />"
                info += f"n: {x}<br />" if is_batched else f"axis 1: {x}<br />"
                info += f"val: {str(val)}<br />"
                text[y].append(info)

        layout = dict(
            height=300 if 300 < 80*array.shape[0] else 80*array.shape[0],
            width=500,
            margin=dict(l=0, r=0, t=0, b=0),
            showlegend=False,
            xaxis=dict(
                title="n" if is_batched else "axis 1",
                showgrid=True,
                showline=True,
            ),
            yaxis=dict(
                title="batch" if is_batched else "axis 0",
                autorange="reversed",
                showgrid=True,
                showticklabels=False,
            )
        )
        fig = go.Figure(
            data=go.Heatmap(
                z=abs(array[:20]), 
                colorscale="viridis", 
                showscale=False,
                hoverinfo="text",
                text=text,
            ), 
            layout=layout
        )
